# FiveM VPS Komutları - Hızlı Referans

## VPS'e Bağlanma
ssh root@194.105.5.37

## Sunucu Yönetimi
systemctl start fivem      # Başlat
systemctl stop fivem       # Durdur
systemctl restart fivem    # Yeniden başlat
systemctl status fivem     # Durum kontrol

## Log Görüntüleme
journalctl -u fivem -f     # Canlı log
journalctl -u fivem -n 100 # Son 100 satır

## Screen Kullanımı (Manuel başlatma)
screen -S fivem            # Yeni screen
screen -r fivem            # Screen'e dön
CTRL+A sonra D             # Screen'den çık

## Veritabanı
mysql -u root -p           # MySQL'e giriş
mysql -u fivem -p temelPaket  # FiveM DB'ye giriş

## Dosya Düzenleme
nano ~/fivem-server/data/server.cfg  # Server.cfg düzenle

## Cache Temizleme
rm -rf ~/fivem-server/data/cache/files/*

## Yedekleme
mysqldump -u fivem -p temelPaket > backup.sql
tar -czf backup.tar.gz ~/fivem-server/data

## Port Kontrolü
netstat -tulpn | grep 30120

## Sistem Bilgisi
htop                       # CPU/RAM
df -h                      # Disk kullanımı
free -h                    # RAM kullanımı

## Güvenlik Duvarı
ufw status                 # Durum
ufw allow 30120/tcp        # Port aç
ufw enable                 # Etkinleştir

## Dosya İzinleri
chmod +x ~/fivem-server/run.sh
chown -R root:root ~/fivem-server

## Sunucu Güncelleme
cd ~/fivem-server
wget https://runtime.fivem.net/artifacts/fivem/build_proot_linux/master/latest/fx.tar.xz
systemctl stop fivem
tar xf fx.tar.xz
rm fx.tar.xz
systemctl start fivem

## Sorun Giderme
# MySQL bağlantı testi
mysql -u fivem -p -e "USE temelPaket; SHOW TABLES;"

# Port dinleme kontrolü
ss -tulpn | grep 30120

# Disk alanı kontrolü
du -sh ~/fivem-server/*

# Log dosyası boyutu
du -sh /var/log/journal/

## Hızlı Restart
systemctl restart fivem && journalctl -u fivem -f
